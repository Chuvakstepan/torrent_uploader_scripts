// ==UserScript==
// @name         RuTracker.org to Torrent Upload
// @namespace    http://tampermonkey.net/
// @version      1.04
// @description  Добавляет кнопку для отправки торрент-файла на локальный сервер с загрузкой постеров
// @author       Chuvakstepan, DeepSeek
// @match        https://rutracker.org/forum/viewtopic.php?t=*
// @grant        GM_xmlhttpRequest
// @grant        GM_download
// @connect      192.168.1.133
// @connect      rutracker.org
// @connect      static.rutracker.cc
// @connect      fastpic.org
// @connect      i127.fastpic.org
// @connect      i4.imageban.ru
// @connect      imageban.ru
// @connect      images4.imagebam.com
// @connect      imagebam.com
// @run-at       document-end
// ==/UserScript==

(function() {
    'use strict';

    console.log('=== Скрипт RuTracker.org Uploader запущен ===');

    const LOCAL_POSTER_SCRIPT = 'http://192.168.1.133/postimg.php';

    // Функция для получения ID раздачи из URL
    function getTorrentId() {
        const urlParams = new URLSearchParams(window.location.search);
        return urlParams.get('t');
    }

    // Функция для получения заголовка раздачи
    function getTorrentTitle() {
        const pageTitle = document.querySelector('title');
        if (pageTitle) {
            let title = pageTitle.textContent.trim();
            title = title.replace(/:: RuTracker\.org$/, '').trim();
            if (title) {
                console.log('✅ Title из title страницы:', title);
                return title;
            }
        }

        const h1Title = document.querySelector('#topic-title');
        if (h1Title) {
            const title = h1Title.textContent.trim();
            if (title) {
                console.log('✅ Title из h1:', title);
                return title;
            }
        }

        console.log('❌ Title не найден');
        return '';
    }

    // Функция для получения URL постера (расширенная)
    function getPosterUrl() {
        // Ищем изображения в сообщении
        const images = document.querySelectorAll('.post_body img');
        
        // Список хостов для поиска
        const posterHosts = ['fastpic', 'imageban', 'imagebam', 'postimg', 'imgur', 'images4'];
        
        // Сначала ищем большие изображения
        for (const img of images) {
            let src = img.src;
            if (!src || !src.startsWith('http')) continue;
            
            // Проверяем, что это хост для изображений
            const isPosterHost = posterHosts.some(host => src.toLowerCase().includes(host));
            if (!isPosterHost) continue;
            
            // Проверяем размеры
            const width = img.naturalWidth || img.width || 0;
            const height = img.naturalHeight || img.height || 0;
            
            // Если изображение большое - это скорее всего постер
            if (width > 200 || height > 200 || (width === 0 && height === 0)) {
                console.log('✅ Постер найден (размер ' + width + 'x' + height + '):', src);
                return src;
            }
        }
        
        // Если не нашли большое, берем первое подходящее
        for (const img of images) {
            let src = img.src;
            if (!src || !src.startsWith('http')) continue;
            
            const isPosterHost = posterHosts.some(host => src.toLowerCase().includes(host));
            if (isPosterHost) {
                console.log('✅ Постер найден (любое):', src);
                return src;
            }
        }

        console.log('❌ Постер не найден');
        return '';
    }

    // Функция для получения URL скачивания торрента
    function getDownloadUrl() {
        const downloadLink = document.querySelector('a.dl-link[href*="dl.php?t="]');
        if (downloadLink) {
            const url = downloadLink.href;
            console.log('✅ Ссылка на торрент найдена:', url);
            return url;
        }
        console.log('❌ Ссылка на торрент не найдена');
        return null;
    }

    // Функция для скачивания торрента через GM_xmlhttpRequest
    function downloadTorrentGM(url) {
        return new Promise((resolve, reject) => {
            console.log('📥 Скачивание торрента с URL:', url);

            GM_xmlhttpRequest({
                method: 'GET',
                url: url,
                responseType: 'blob',
                onload: function(response) {
                    console.log('📥 Ответ получен, статус:', response.status);
                    if (response.status === 200) {
                        console.log('📥 Торрент скачан, размер:', response.response.size, 'байт');
                        resolve(response.response);
                    } else {
                        reject(new Error(`HTTP ${response.status}: ${response.statusText}`));
                    }
                },
                onerror: function(error) {
                    console.error('❌ Ошибка при скачивании торрента:', error);
                    reject(new Error('Ошибка сети при скачивании торрента'));
                },
                ontimeout: function() {
                    reject(new Error('Таймаут при скачивании торрента'));
                },
                timeout: 30000
            });
        });
    }

    // Функция: Загрузка постера на локальный сервер
    function uploadPosterToLocal(posterUrl) {
        return new Promise((resolve) => {
            console.log('🖼️ Загрузка постера на локальный сервер:', posterUrl);

            // Кодируем URL правильно
            const encodedUrl = encodeURIComponent(posterUrl);
            const requestUrl = LOCAL_POSTER_SCRIPT + '?url=' + encodedUrl;

            console.log('🔄 Запрос к:', requestUrl);

            GM_xmlhttpRequest({
                method: 'GET',
                url: requestUrl,
                onload: function(response) {
                    console.log('📥 Ответ от postimg.php: статус', response.status);
                    console.log('📥 Тело ответа:', response.responseText);

                    if (response.status === 200) {
                        try {
                            const jsonResponse = JSON.parse(response.responseText);
                            console.log('📥 Распарсенный ответ:', jsonResponse);

                            if (jsonResponse.success && jsonResponse.url) {
                                console.log('✅ Постер загружен локально:', jsonResponse.url);
                                resolve(jsonResponse.url);
                            } else if (jsonResponse.error) {
                                console.warn('⚠️ Ошибка загрузки постера:', jsonResponse.error);
                                resolve(posterUrl);
                            } else {
                                console.warn('⚠️ Неизвестный ответ, используем оригинальный постер');
                                resolve(posterUrl);
                            }
                        } catch (e) {
                            console.warn('⚠️ Ошибка парсинга JSON:', e.message);
                            console.warn('⚠️ Ответ не является JSON, используем оригинальный постер');
                            resolve(posterUrl);
                        }
                    } else {
                        console.warn('⚠️ Локальный сервер вернул статус ' + response.status + ', используем оригинальный постер');
                        resolve(posterUrl);
                    }
                },
                onerror: function(error) {
                    console.warn('⚠️ Ошибка соединения с локальным сервером, используем оригинальный постер:', error);
                    resolve(posterUrl);
                },
                ontimeout: function() {
                    console.warn('⚠️ Таймаут соединения с локальным сервером, используем оригинальный постер');
                    resolve(posterUrl);
                },
                timeout: 15000 // Таймаут 15 секунд
            });
        });
    }

    // Функция для отправки торрента на сервер
    function uploadTorrent(blob, torrentName, title, posterUrl) {
        return new Promise((resolve, reject) => {
            const formData = new FormData();
            formData.append('file', blob, torrentName);
            formData.append('save', 'true');
            if (title) {
                formData.append('title', title);
            }
            if (posterUrl) {
                formData.append('poster', posterUrl);
            }

            console.log('📤 ОТПРАВКА ЗАПРОСА:');
            console.log('📤 URL:', 'http://192.168.1.133:8090/torrent/upload');
            console.log('📤 Метод: POST');
            console.log('📤 Параметры формы:');
            console.log('  file:', torrentName, `(${blob.size} байт)`);
            console.log('  save: true');
            console.log('  title:', title || '(пусто)');
            console.log('  poster:', posterUrl || '(пусто)');

            GM_xmlhttpRequest({
                method: 'POST',
                url: 'http://192.168.1.133:8090/torrent/upload',
                headers: {
                    'Accept': 'application/json'
                },
                data: formData,
                onload: function(response) {
                    console.log('📥 ОТВЕТ СЕРВЕРА:');
                    console.log('📥 Статус:', response.status);
                    console.log('📥 Тело ответа:', response.responseText);

                    if (response.status === 200 || response.status === 201) {
                        try {
                            const jsonResponse = JSON.parse(response.responseText);
                            console.log('✅ Ответ сервера (JSON):', jsonResponse);
                            resolve(jsonResponse);
                        } catch (e) {
                            console.log('✅ Ответ сервера (текст):', response.responseText);
                            resolve(response.responseText);
                        }
                    } else {
                        reject(new Error(`HTTP ${response.status}: ${response.statusText}`));
                    }
                },
                onerror: function(error) {
                    console.error('❌ Ошибка отправки на сервер:', error);
                    reject(new Error('Ошибка сети при отправке на сервер'));
                },
                ontimeout: function() {
                    reject(new Error('Таймаут при отправке на сервер'));
                },
                timeout: 30000
            });
        });
    }

    // Основная функция
    function addUploadButton() {
        console.log('🔍 Поиск контейнера для кнопки...');

        const torrentContainer = document.querySelector('td[style*="padding: 10px 6px;"]');
        if (!torrentContainer) {
            console.log('❌ Контейнер с кнопкой скачивания не найден');
            return;
        }

        if (document.getElementById('rutracker-upload-container')) {
            console.log('⚠️ Кнопка уже существует');
            return;
        }

        const torrentId = getTorrentId();
        const torrentTitle = getTorrentTitle();
        const posterUrl = getPosterUrl();
        const downloadUrl = getDownloadUrl();

        console.log('📋 Данные раздачи:');
        console.log('  ID:', torrentId);
        console.log('  Title:', torrentTitle || '(пусто)');
        console.log('  Poster URL:', posterUrl || '(пусто)');
        console.log('  Download URL:', downloadUrl);

        if (!torrentId || !downloadUrl) {
            console.log('❌ Не удалось получить ID раздачи или URL скачивания');
            return;
        }

        const buttonContainer = document.createElement('div');
        buttonContainer.id = 'rutracker-upload-container';
        buttonContainer.style.cssText = 'padding-top: 8px; text-align: center;';

        const uploadButton = document.createElement('input');
        uploadButton.id = 'rutracker-upload-button';
        uploadButton.type = 'button';
        uploadButton.className = 'lite';
        uploadButton.style.cssText = 'width: 120px;';
        uploadButton.value = '📤 Отправить';

        const statusDiv = document.createElement('div');
        statusDiv.id = 'rutracker-upload-status';
        statusDiv.style.cssText = `
            margin-top: 5px;
            font-size: 12px;
            transition: opacity 0.5s ease;
            display: none;
            padding: 3px 8px;
            border-radius: 3px;
            max-width: 200px;
            margin-left: auto;
            margin-right: auto;
            box-sizing: border-box;
            word-wrap: break-word;
        `;

        buttonContainer.appendChild(uploadButton);
        buttonContainer.appendChild(statusDiv);
        torrentContainer.appendChild(buttonContainer);

        console.log('✅ Кнопка добавлена');

        uploadButton.addEventListener('click', async function() {
            const statusDiv = document.getElementById('rutracker-upload-status');
            statusDiv.style.display = 'block';
            statusDiv.style.opacity = '1';
            statusDiv.innerHTML = '⏳ Скачивание торрента...';
            statusDiv.style.color = '#333';
            statusDiv.style.background = '#fff3cd';

            uploadButton.disabled = true;
            uploadButton.value = '⏳ Загрузка...';

            try {
                const blob = await downloadTorrentGM(downloadUrl);
                console.log('✅ Торрент успешно скачан, размер:', blob.size);

                statusDiv.innerHTML = '⏳ Обработка постера...';
                statusDiv.style.background = '#cce5ff';

                let finalPosterUrl = posterUrl;
                if (posterUrl) {
                    finalPosterUrl = await uploadPosterToLocal(posterUrl);
                    if (finalPosterUrl === posterUrl) {
                        console.log('ℹ️ Используется оригинальный постер с внешнего сервера');
                    } else {
                        console.log('✅ Постер загружен локально:', finalPosterUrl);
                    }
                }

                statusDiv.innerHTML = '⏳ Отправка на сервер...';
                statusDiv.style.background = '#cce5ff';

                const torrentName = `[rutracker.org]t${torrentId}.torrent`;
                const result = await uploadTorrent(blob, torrentName, torrentTitle, finalPosterUrl);

                statusDiv.innerHTML = '✅ Торрент успешно отправлен!';
                statusDiv.style.color = 'green';
                statusDiv.style.background = '#d4edda';

                console.log('✅ Торрент успешно отправлен!', result);

            } catch (error) {
                console.error('❌ Ошибка:', error);
                statusDiv.innerHTML = `❌ ${error.message || 'Неизвестная ошибка'}`;
                statusDiv.style.color = 'red';
                statusDiv.style.background = '#f8d7da';
            } finally {
                uploadButton.disabled = false;
                uploadButton.value = '📤 Отправить';

                setTimeout(() => {
                    statusDiv.style.opacity = '0';
                    setTimeout(() => {
                        statusDiv.style.display = 'none';
                    }, 500);
                }, 5000);
            }
        });
    }

    function init() {
        console.log('🚀 Инициализация скрипта для RuTracker.org...');
        setTimeout(addUploadButton, 2000);
        setTimeout(addUploadButton, 4000);
    }

    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', init);
    } else {
        init();
    }

    const observer = new MutationObserver(function(mutations) {
        if (!document.getElementById('rutracker-upload-container')) {
            console.log('🔄 Кнопка не найдена, пробуем добавить...');
            addUploadButton();
        }
    });
    observer.observe(document.body, { childList: true, subtree: true });

})();
